Source Code

This code is written in Java:

import java.util.Scanner;
public class main {

    public static void main(String[] arg){

        Scanner scanner = new Scanner(System.in);

        double principal,
               rate,
               monthlyContribution,
               monthlyRate,
               amount;

        int years,
            months;

        System.out.println("\n********************************************");
        System.out.println("Welcome to the Compound Interest Calculator!");
        System.out.println("********************************************\n");

        System.out.print("Enter the principal amount: $");
        principal = scanner.nextDouble();

        System.out.print("Enter the interest rate (in %) you are expecting to get on average: ");
        rate = scanner.nextDouble() / 100;
        monthlyRate = rate / 12;

        System.out.print("Enter the amount you plan to contribute monthly: $");
        monthlyContribution = scanner.nextDouble();

        System.out.print("Enter the number of years: ");
        years = scanner.nextInt();
        months = years * 12;

        // Compound interest formula with monthly contribution
        amount = principal * Math.pow(1 + monthlyRate, months) +
                 (monthlyContribution * ((Math.pow(1 + monthlyRate, months) - 1) / monthlyRate));

        System.out.printf("The amount after %d years is: $%,.2f", years, amount);

        scanner.close();
    }
}

/*
Notes: This is a personal compound interest calculator.
        This calculator is a good reference to see what your investment could be worth over time.
        This could be used for retirement planning, 529, Roth IRA, etc.
        A safe rate to use is 7% - 10%. If you are using the S&P 500 as a reference it is around 10-12%
        Adjustments can be made if you want to do it by age range.
*/

        System.out.println("********************************************\n");

        System.out.print("Enter your current age: ");
        age = scanner.nextDouble();

        System.out.print("Enter the estimated Age you wish to retire: ");
        retirementAge = scanner.nextDouble();
        years = retirementAge - age;
        months = years * 12;

        System.out.print("Enter the current amount you have in retirement: $");
        retirementAmount = scanner.nextDouble();

        System.out.print("Enter the amount you plan to contribute monthly: $");
        monthlyContribution = scanner.nextDouble();

        System.out.print("What is the % interest rate (in %) you are expecting to earn: ");
        rate = scanner.nextDouble() / 100;
        monthlyRate = rate / 12;

        // Compound interest formula with monthly contribution
        amount = retirementAmount * Math.pow(1 + monthlyRate, months) +
                (monthlyContribution * ((Math.pow(1 + monthlyRate, months) - 1) / monthlyRate));

        System.out.printf("The amount after %d years is: $%,.2f", years, amount);

        scanner.close();
    }
}

import java.util.Scanner;

public class main {

    public static void main(String[] arg) {

        Scanner scanner = new Scanner(System.in);

        double netIncome,
                rentMortgage,
                utilities,
                gasoline,
                internet,
                insurances,
                groceries,
                phone,
                toiletries,
                debtMin,
                misc,
                expenses,
                remainingIncome,
                eFundMin,
                eFundThreeMonths,
                eFundSixMonths,
                eFundMax;

        System.out.println("\n********************************************");
        System.out.println("Welcome to the Emergency Fund Calculator!");
        System.out.println("********************************************\n");

        System.out.print("Enter your monthly net income: $");
        netIncome = scanner.nextDouble();

        System.out.print("Enter your monthly rent/mortgage amount: $");
        rentMortgage = scanner.nextDouble();

        System.out.print("Enter your average monthly utility bill: $");
        utilities = scanner.nextDouble();

        System.out.print("Enter your average monthly gasoline bill: $");
        gasoline = scanner.nextDouble();

        System.out.print("Enter your average monthly internet bill: $");
        internet = scanner.nextDouble();

        System.out.print("Enter your average monthly insurance bill: $");
        insurances = scanner.nextDouble();

        System.out.print("Enter your average monthly grocery bill: $");
        groceries = scanner.nextDouble();

        System.out.print("Enter your average monthly phone bill: $");
        phone = scanner.nextDouble();

        System.out.print("Enter your average monthly toiletries bill: $");
        toiletries = scanner.nextDouble();

        System.out.print("Enter your average monthly debt minimum payment: $");
        debtMin = scanner.nextDouble();

        System.out.print("Enter your average monthly miscellaneous expense: $");
        misc = scanner.nextDouble();
        System.out.println();

        // Mathematics //
        expenses = rentMortgage + utilities + gasoline + internet + insurances + groceries + phone + toiletries
                + debtMin + misc;

        remainingIncome = netIncome - expenses;

        eFundMin = expenses;
        eFundThreeMonths = expenses * 3;
        eFundSixMonths = expenses * 6;
        eFundMax = expenses * 12;

        System.out.printf("The disposable amount of income is: $%,.2f\n\n", remainingIncome);
        System.out.printf("The minimum amount for an emergency fund is: $%,.2f\n", eFundMin);
        System.out.printf("The medium amount for an emergency fund is: $%,.2f - $%,.2f\n", eFundThreeMonths, eFundSixMonths);
        System.out.printf("The maximum amount for an emergency fund is: $%,.2f\n", eFundMax);
        System.out.printf("Any amount beyond this should be invested in retirement or a brokerage account.\n",
                eFundMax);

        scanner.close();
    }
}
