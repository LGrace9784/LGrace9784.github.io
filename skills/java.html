// Java OOP Example - Calculator Class
public class Calculator {
    private double result;

    public Calculator() {
        this.result = 0.0;
    }

    public Calculator add(double number) {
        this.result += number;
        return this;
    }

    public Calculator subtract(double number) {
        this.result -= number;
        return this;
    }

    public Calculator multiply(double number) {
        this.result *= number;
        return this;
    }

    public Calculator divide(double number) {
        if (number != 0) {
            this.result /= number;
        } else {
            throw new IllegalArgumentException("Cannot divide by zero");
        }
        return this;
    }

    public double getResult() {
        return this.result;
    }

    public Calculator clear() {
        this.result = 0.0;
        return this;
    }

    // Method overloading example
    public Calculator add(int a, int b) {
        this.result = a + b;
        return this;
    }

    public Calculator add(double a, double b) {
        this.result = a + b;
        return this;
    }
}

// Usage example with method chaining
public class Main {
    public static void main(String[] args) {
        Calculator calc = new Calculator();
        double result = calc.add(10).multiply(2).subtract(5).getResult();
        System.out.println("Result: " + result); // Output: 15.0
    }
}
